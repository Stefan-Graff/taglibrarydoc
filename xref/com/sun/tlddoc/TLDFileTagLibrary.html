32  package com.sun.tlddoc;
33  
34  import java.io.File;
35  import java.io.FileInputStream;
36  import java.io.IOException;
37  import java.io.InputStream;
38  import javax.xml.parsers.DocumentBuilder;
39  import javax.xml.transform.TransformerException;
40  import org.w3c.dom.Document;
41  import org.xml.sax.InputSource;
42  import org.xml.sax.SAXException;
43  
44  /**
45   * Tag library represented by a single standalone TLD file
46   *
47   * @author  mroth
48   */
49  public class TLDFileTagLibrary
50      extends TagLibrary
51  {
52  
53      /**
54       * The location of the TLD file for this tag library
55       */
56      final private File tldFile;
57  
58      /**
59       * Creates a new instance of {@link TLDFileTagLibrary}
60       *
61       * @param tldFile location of the TLD file for this tag library
62       */
63      public TLDFileTagLibrary( File tldFile ) {
64          this.tldFile = tldFile;
65      }
66  
67      /**
68       * {@inheritDoc}
69       */
70      @Override
71      public String getPathDescription() {
72          return this.tldFile.getAbsolutePath();
73      }
74  
75      /**
76       * {@inheritDoc}
77       */
78      @Override
79      public Document getTLDDocument(DocumentBuilder documentBuilder)
80          throws IOException, SAXException, TransformerException
81      {
82          FileInputStream in = new FileInputStream( this.tldFile );
83          InputSource source;
84          Document result;
85          try {
86              source = new InputSource( in );
87              result = documentBuilder.parse( source );
88          }
89          finally {
90              in.close();
91          }
92          return result;
93      }
94  
95      /**
96       * {@inheritDoc}
97       */
98      @Override
99      public InputStream getResource(String path)
100         throws IOException
101     {
102         InputStream result = null;
103 
104         // This is a bit of a guess, since we don't know where the TLD is.
105         // Start from the directory containing the TLD, and backtrack,
106         // using the path as a relative path.
107         //   For example:
108         //      TLD:  /home/mroth/test/sample/WEB-INF/tld/test.tld
109         //      path: /WEB-INF/tags/tag1.tag
110 
111         File dir = tldFile.getParentFile();
112         if( path.startsWith( "/" ) ) {
113             path = path.substring( 1 );
114         }
115         File look = null;
116         while( dir != null && !(look = new File( dir, path )).exists() ) {
117             dir = dir.getParentFile();
118         }
119 
120         if( look != null && look.exists() ) {
121             // Found it (or something pretty close to it anyway)
122             result = new FileInputStream( look );
123         }
124 
125         return result;
126     }
127 
128     /**
129      * {@inheritDoc}
130      */
131     @Override
132     public void close() throws IOException {
133         // Nothing to do
134     }
135 }
